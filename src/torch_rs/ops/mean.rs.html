1use super::Op;
2use crate::tensor::Tensor;
3use ndarray::{Array, ArrayD, IxDyn};
4use std::rc::Rc;
5
6#[derive(Debug)]
7pub struct Mean {
8    /// The shape of the input tensor, saved for the backward pass.
9    input_shape: Vec<usize>,
10}
11
12impl Mean {
13    pub fn new(input_shape: Vec<usize>) -> Self {
14        Mean { input_shape }
15    }
16}
17
18impl Op for Mean {
19    /// Computes the mean of a tensor.
20    fn forward(&self, inputs: &[&Tensor]) -> Tensor {
21        if inputs.len() != 1 {
22            panic!("Mean operation takes exactly one input.");
23        }
24        let input = &inputs[0];
25        let data = &input.0.borrow().data;
26
27        // Calculate the mean. ndarray's mean() returns Option<f32>.
28        let mean_val = data
29            .mean()
30            .expect("Cannot compute mean of an empty tensor.");
31
32        // Create a true scalar (0-dim) tensor
33        let result_data = Array::from_elem(IxDyn(&[]), mean_val).into_dyn();
34        let result = Tensor::new(result_data);
35
36        // Set up the computation graph for backpropagation.
37        if input.0.borrow().requires_grad {
38            // For a scalar tensor, we need a scalar gradient
39            result.0.borrow_mut().grad = Some(ArrayD::from_elem(IxDyn(&[]), 1.0));
40            let op = Mean::new(input.shape());
41            result.0.borrow_mut().set_creator(Rc::new(op));
42            result.0.borrow_mut().add_parent(input);
43            result.0.borrow_mut().requires_grad = true;
44        } else {
45            // Explicitly set requires_grad to false if input doesn't require grad
46            result.0.borrow_mut().requires_grad = false;
47        }
48        result
49    }
50
51    /// Computes the gradient of the mean operation.
52    fn backward(&self, parent: &Tensor) -> Vec<ArrayD<f32>> {
53        // The gradient from the output tensor (which is the parent in the graph)
54        let grad_output = parent
55            .0
56            .borrow()
57            .grad
58            .as_ref()
59            .expect("Gradient not found in backward pass")
60            .clone();
61
62        let grad_output_scalar = grad_output.iter().next().cloned().unwrap_or(1.0);
63
64        // The number of elements in the original input tensor.
65        let num_elements = self.input_shape.iter().product::<usize>() as f32;
66        if num_elements == 0.0 {
67            return vec![ArrayD::zeros(IxDyn(&self.input_shape))];
68        }
69
70        // The gradient for each element of the input is grad_output / N.
71        let grad_val = grad_output_scalar / num_elements;
72
73        // Create the gradient array for the input, with the same shape as the original input.
74        let grad_input = ArrayD::from_elem(IxDyn(&self.input_shape), grad_val);
75
76        vec![grad_input]
77    }
78}
79
80/// Functional interface for the mean operation.
81pub fn mean(tensor: &Tensor) -> Tensor {
82    let op = Mean::new(tensor.shape());
83    op.forward(&[tensor])
84}
85
86impl Tensor {
87    /// Computes the mean of the tensor.
88    pub fn mean(&self) -> Tensor {
89        mean(self)
90    }
91}
