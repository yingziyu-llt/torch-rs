7#[derive(Debug)]
8pub struct Multiply {
9    input_shapes: Vec<Vec<usize>>,
10    a_data: Option<ndarray::ArrayD<f32>>,
11    b_data: Option<ndarray::ArrayD<f32>>,
12}
13
14impl Multiply {
15    pub fn new(input_shapes: Vec<Vec<usize>>) -> Self {
16        Multiply {
17            input_shapes,
18            a_data: None,
19            b_data: None,
20        }
21    }
22
23    fn get_broadcasted_shape(shape1: &[usize], shape2: &[usize]) -> Vec<usize> {
24        let len1 = shape1.len();
25        let len2 = shape2.len();
26        let max_len = std::cmp::max(len1, len2);
27
28        let mut result = Vec::with_capacity(max_len);
29
30        // 从尾部开始，依次比较对齐的维度
31        for i in 0..max_len {
32            let dim1 = if i < len1 { shape1[len1 - 1 - i] } else { 1 };
33            let dim2 = if i < len2 { shape2[len2 - 1 - i] } else { 1 };
34
35            if dim1 != 1 && dim2 != 1 && dim1 != dim2 {
36                panic!(
37                    "Incompatible dimensions for broadcasting: {} and {}",
38                    dim1, dim2
39                );
40            }
41
42            result.push(std::cmp::max(dim1, dim2));
43        }
44
45        result.reverse();
46        result
47    }
48
49    /// 将张量按目标形状进行求和的辅助函数。
50    /// 用于广播操作的反向传播。
51    fn sum_to_shape(grad: &ArrayD<f32>, target_shape: &[usize]) -> ArrayD<f32> {
52        let mut result = grad.clone();
53        let grad_shape = grad.shape();
54
55        // 1. 处理维度数不同的情况 (对多余的前导维度求和)
56        if target_shape.len() < grad_shape.len() {
57            let axes_to_sum: Vec<usize> = (0..(grad_shape.len() - target_shape.len())).collect();
58            for &axis in axes_to_sum.iter().rev() {
59                result = result.sum_axis(Axis(axis));
60            }
61        }
62
63        // 2. 处理被广播的维度 (size 1 -> size > 1)
64        let mut sum_axes = Vec::new();
65        let current_shape = result.shape(); // 获取求和后的新形状
66        for i in 0..target_shape.len() {
67            // 如果目标维度是1，但当前梯度维度大于1，说明发生了广播
68            if target_shape[i] == 1 && current_shape[i] > 1 {
69                sum_axes.push(i);
70            }
71        }
72
73        // 从大到小排序轴，以避免求和后索引变化
74        sum_axes.sort_unstable_by(|a, b| b.cmp(a));
75        for axis in sum_axes {
76            result = result.sum_axis(Axis(axis));
77        }
78
79        // 3. 确保形状完全匹配 (恢复被求和掉的 size 1 的维度)
80        if result.shape() != target_shape {
81            result = result.into_shape(IxDyn(target_shape)).unwrap();
82        }
83
84        result
85    }
86}

88impl Op for Multiply {
89    fn forward(&self, inputs: &[&Tensor]) -> Tensor {
90        // 逐元素相乘，广播机制
91        let a = &inputs[0].0.borrow().data;
92        let b = &inputs[1].0.borrow().data;
93
94        let result = match (a.broadcast(b.shape()), b.broadcast(a.shape())) {
95            (Some(a_broadcast), _) => &a_broadcast * b,
96            (_, Some(b_broadcast)) => a * &b_broadcast,
97            // 如果两个都不能直接广播，尝试计算共同的广播形状
98            _ => {
99                let shape1 = a.shape();
100                let shape2 = b.shape();
101                let broadcast_shape = Self::get_broadcasted_shape(shape1, shape2);
102                let a_broadcasted = a.broadcast(IxDyn(&broadcast_shape)).unwrap_or_else(|| {
103                    panic!(
104                        "Cannot broadcast tensor with shape {:?} to {:?}",
105                        shape1, broadcast_shape
106                    )
107                });
108                let b_broadcasted = b.broadcast(IxDyn(&broadcast_shape)).unwrap_or_else(|| {
109                    panic!(
110                        "Cannot broadcast tensor with shape {:?} to {:?}",
111                        shape2, broadcast_shape
112                    )
113                });
114                a_broadcasted.to_owned() * b_broadcasted.to_owned()
115            }
116        };
117
118        let op = Rc::new(Multiply {
119            input_shapes: self.input_shapes.clone(),
120            a_data: Some(a.clone()),
121            b_data: Some(b.clone()),
122        });
123
124        let result_tensor = Tensor::new(result);
125        result_tensor.0.borrow_mut().set_creator(op);
126        result_tensor.0.borrow_mut().add_parent(&inputs[0]);
127        result_tensor.0.borrow_mut().add_parent(&inputs[1]);
128        result_tensor.0.borrow_mut().requires_grad =
129            inputs[0].0.borrow().requires_grad || inputs[1].0.borrow().requires_grad;
130        result_tensor
131    }
132    fn backward(&self, parent: &Tensor) -> Vec<ArrayD<f32>> {
133        let grad_output = parent
134            .0
135            .borrow()
136            .grad
137            .as_ref()
138            .expect("Parent gradient is None")
139            .clone();
140        let a_data = self.a_data.as_ref().expect("a_data is None in backward");
141        let b_data = self.b_data.as_ref().expect("b_data is None in backward");
142
143        // 相对于 a 的梯度: ∂L/∂a = ∂L/∂c * b
144        let mut grad_a = &grad_output * b_data;
145        // 如果发生了广播，需要将梯度求和到原始形状
146        if grad_a.shape() != a_data.shape() {
147            grad_a = Self::sum_to_shape(&grad_a, a_data.shape());
148        }
149
150        // 相对于 b 的梯度: ∂L/∂b = ∂L/∂c * a
151        let mut grad_b = &grad_output * a_data;
152        // 如果发生了广播，需要将梯度求和到原始形状
153        if grad_b.shape() != b_data.shape() {
154            grad_b = Self::sum_to_shape(&grad_b, b_data.shape());
155        }
156
157        vec![grad_a, grad_b]
158    }
159}

161impl<'a, 'b> Mul<&'a Tensor> for &'b Tensor {
162    type Output = Tensor;
163
164    fn mul(self, other: &'a Tensor) -> Tensor {
165        let op = Multiply::new(vec![self.shape().to_vec(), other.shape().to_vec()]);
166        let inputs = vec![self, other];
167        op.forward(&inputs)
168    }
169}
