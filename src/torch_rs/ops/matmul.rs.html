1use super::Op;
2use crate::tensor::Tensor;
3use core::panic;
4use ndarray::{ArrayD, Axis, Ix2};
5use ndarray_einsum::tensordot;
6use std::rc::Rc;

8#[derive(Debug)]
9pub struct MatMul {
10    input_shapes: Vec<Vec<usize>>,
11    a_data: Option<ArrayD<f32>>,
12    b_data: Option<ArrayD<f32>>,
13}
14
15impl MatMul {
16    pub fn new() -> Self {
17        MatMul {
18            input_shapes: Vec::new(),
19            a_data: None,
20            b_data: None,
21        }
22    }
23}
24
25impl Op for MatMul {
26    fn forward(&self, inputs: &[&Tensor]) -> Tensor {
27        // A: [batch, input] or [batch, channel, input]
28        // B: [input, output]
29        // Output: [batch, output] or [batch, channel, output]
30        if inputs.len() != 2 {
31            panic!("MatMul requires exactly two input tensors");
32        }
33        let a = &inputs[0].0.borrow().data;
34        let b = &inputs[1].0.borrow().data;
35
36        if b.ndim() != 2 {
37            panic!("Input B must be a 2D tensor (weights), got {}D", b.ndim());
38        }
39
40        let a_ndim = a.ndim();
41        if a_ndim != 2 && a_ndim != 3 {
42            panic!("Input A must be a 2D or 3D tensor, got {}D", a_ndim);
43        }
44
45        let a_last_dim_idx = a_ndim - 1;
46        let a_last_dim_size = a.shape()[a_last_dim_idx];
47        let b_first_dim_size = b.shape()[0];
48
49        if a_last_dim_size != b_first_dim_size {
50            panic!(
51                "Matrix dimensions do not match for multiplication. A's last dim ({}) is {} and B's first dim ({}) is {}",
52                a_last_dim_idx, a_last_dim_size, 0, b_first_dim_size
53            );
54        }
55
56        let output_data = if a_ndim == 2 {
57            // A: [batch, input], B: [input, output] -> Output: [batch, output]
58            let a_2d = a.view().into_dimensionality::<Ix2>().unwrap();
59            let b_2d = b.view().into_dimensionality::<Ix2>().unwrap();
60            a_2d.dot(&b_2d).into_dyn()
61        } else {
62            // a_ndim == 3
63            // A: [batch, channel, input], B: [input, output] -> Output: [batch, channel, output]
64            tensordot(a, b, &[Axis(2)], &[Axis(0)]).into_dyn()
65        };
66
67        let result = Tensor::new(output_data);
68        if inputs[0].0.borrow().requires_grad || inputs[1].0.borrow().requires_grad {
69            result.0.borrow_mut().grad = Some(ArrayD::zeros(result.shape()));
70            let op = MatMul {
71                input_shapes: vec![inputs[0].shape().to_vec(), inputs[1].shape().to_vec()],
72                a_data: Some(a.clone()),
73                b_data: Some(b.clone()),
74            };
75            result.0.borrow_mut().set_creator(Rc::new(op));
76        }
77        if inputs[0].0.borrow().requires_grad {
78            result.0.borrow_mut().add_parent(inputs[0]);
79        }
80        if inputs[1].0.borrow().requires_grad {
81            result.0.borrow_mut().add_parent(inputs[1]);
82        }
83        result.0.borrow_mut().requires_grad =
84            inputs[0].0.borrow().requires_grad || inputs[1].0.borrow().requires_grad;
85        result
86    }
87
88    fn backward(&self, grad: &Tensor) -> Vec<ArrayD<f32>> {
89        let grad_output = grad
90            .0
91            .borrow()
92            .grad
93            .as_ref()
94            .expect("Gradient not found in backward pass")
95            .clone();
96        let a = self.a_data.as_ref().expect("a_data not saved in MatMul");
97        let b = self.b_data.as_ref().expect("b_data not saved in MatMul");
98
99        let a_ndim = a.ndim();
100        let b_2d = b.view().into_dimensionality::<Ix2>().unwrap();
101
102        // --- Calculate grad_a = grad_output @ b.T ---
103        let b_t = b_2d.t();
104        let grad_a = if a_ndim == 2 {
105            let grad_output_2d = grad_output.view().into_dimensionality::<Ix2>().unwrap();
106            grad_output_2d.dot(&b_t).into_dyn()
107        } else {
108            // a_ndim == 3
109            // grad_output: [B, C, N], b_t: [N, K] -> grad_a: [B, C, K]
110            tensordot(&grad_output, &b_t, &[Axis(2)], &[Axis(0)]).into_dyn()
111        };
112
113        // --- Calculate grad_b = a.T @ grad_output ---
114        // This involves summing over the batch dimension (and channel if it exists).
115        let grad_b = if a_ndim == 2 {
116            // a: [B, K], grad_output: [B, N] -> grad_b: [K, N]
117            let a_2d = a.view().into_dimensionality::<Ix2>().unwrap();
118            let grad_output_2d = grad_output.view().into_dimensionality::<Ix2>().unwrap();
119            a_2d.t().dot(&grad_output_2d).into_dyn()
120        } else {
121            // a_ndim == 3
122            // a: [B, C, K], grad_output: [B, C, N] -> grad_b: [K, N]
123            // We need to contract the first two axes (batch and channel).
124            tensordot(a, &grad_output, &[Axis(0), Axis(1)], &[Axis(0), Axis(1)]).into_dyn()
125        };
126
127        vec![grad_a, grad_b]
128    }
129}
130
131pub fn matmul(a: &Tensor, b: &Tensor) -> Tensor {
132    let opt = MatMul::new();
133    opt.forward(&[a, b])
134}
