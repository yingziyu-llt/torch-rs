1use super::Op;
2use crate::tensor::Tensor;
3use ndarray::{ArrayD, Axis, IxDyn};
4use std::ops::Add;
5use std::rc::Rc;

7#[derive(Debug)]
8pub struct AddOp {
9    input_shapes: Vec<Vec<usize>>, // 存储输入张量的形状
10}
11
12impl AddOp {
13    pub fn new(shape1: Vec<usize>, shape2: Vec<usize>) -> Self {
14        AddOp {
15            input_shapes: vec![shape1, shape2],
16        }
17    }
18    
19    // 获取广播后的形状
20    fn get_broadcasted_shape(shape1: &[usize], shape2: &[usize]) -> Vec<usize> {
21        let len1 = shape1.len();
22        let len2 = shape2.len();
23        let max_len = std::cmp::max(len1, len2);
24        
25        let mut result = Vec::with_capacity(max_len);
26        
27        // 从尾部开始，依次比较对齐的维度
28        for i in 0..max_len {
29            let dim1 = if i < len1 { shape1[len1 - 1 - i] } else { 1 };
30            let dim2 = if i < len2 { shape2[len2 - 1 - i] } else { 1 };
31            
32            if dim1 != 1 && dim2 != 1 && dim1 != dim2 {
33                panic!("Incompatible dimensions for broadcasting: {} and {}", dim1, dim2);
34            }
35            
36            result.push(std::cmp::max(dim1, dim2));
37        }
38        
39        result.reverse();
40        result
41    }
42}

44impl Op for AddOp {
45    /// 前向传播：计算输入张量的加法结果
46    fn forward(&self, inputs: &[&Tensor]) -> Tensor {
47        if inputs.len() != 2 {
48            panic!("AddOp requires exactly two input tensors");
49        }
50
51        let a = &inputs[0].0.borrow().data;
52        let b = &inputs[1].0.borrow().data;
53        
54        // 使用ndarray的广播机制处理张量加法
55        let result_data = match (a.broadcast(b.shape()), b.broadcast(a.shape())) {
56            (Some(a_broadcast), _) => &a_broadcast + b,
57            (_, Some(b_broadcast)) => a + &b_broadcast,
58            // 如果两个都不能直接广播，尝试计算共同的广播形状
59            _ => {
60                let shape1 = a.shape();
61                let shape2 = b.shape();
62                let output_shape = Self::get_broadcasted_shape(shape1, shape2);
63                
64                let a_broadcast = a.broadcast(IxDyn(&output_shape))
65                    .expect("Failed to broadcast first tensor");
66                let b_broadcast = b.broadcast(IxDyn(&output_shape))
67                    .expect("Failed to broadcast second tensor");
68                
69                &a_broadcast.to_owned() + &b_broadcast
70            }
71        };
72
73        // 创建一个新的op实例，存储输入形状供反向传播使用
74        let op = AddOp::new(a.shape().to_vec(), b.shape().to_vec());
75        
76        // 创建新的张量作为输出
77        let output = Tensor::new(result_data);
78        {
79            let mut output_data = output.0.borrow_mut();
80            output_data.set_creator(Rc::new(op));
81
82            // 添加需要梯度的父节点
83            if inputs[0].0.borrow().requires_grad {
84            output_data.add_parent(inputs[0]);
85            }
86            
87            if inputs[1].0.borrow().requires_grad {
88            output_data.add_parent(inputs[1]);
89            }
90            
91            // 如果任一输入需要梯度，则输出也需要梯度
92            output_data.requires_grad = inputs[0].0.borrow().requires_grad || inputs[1].0.borrow().requires_grad;
93        }
94        
95        output
96    }
97
98    fn backward(&self, parent: &Tensor) -> Vec<ArrayD<f32>> {
99        let grad = parent.0.borrow().grad.as_ref().expect("Gradient should not be None").clone();
100        let mut grads = Vec::with_capacity(2);
101        // 对于每个输入形状，处理梯度的反向广播
102        for shape in &self.input_shapes {
103            let mut result = grad.clone();
104            // 先对多余的维度 sum
105            while result.ndim() > shape.len() {
106                result = result.sum_axis(Axis(0));
107            }
108            // 再对 shape=1 的维度 sum，倒序处理 axis
109            let result_shape: Vec<usize> = result.shape().to_vec();
110            let mut sum_axes = Vec::new();
111            for (i, (&s, &g)) in shape.iter().zip(result_shape.iter()).enumerate() {
112                if s == 1 && g > 1 {
113                    sum_axes.push(i);
114                }
115            }
116            sum_axes.sort_unstable_by(|a, b| b.cmp(a));
117            for axis in sum_axes {
118                result = result.sum_axis(Axis(axis));
119            }
120            // 确保形状完全匹配
121            if result.shape() != shape.as_slice() {
122                result = result.into_shape(IxDyn(shape)).unwrap();
123            }
124            grads.push(result);
125        }
126        grads
127    }
128}

130impl<'a, 'b> Add<&'a Tensor> for &'b Tensor {
131    type Output = Tensor;
132
133    fn add(self, other: &'a Tensor) -> Tensor {
134        // 直接使用输入形状创建op
135        let op = AddOp::new(
136            self.0.borrow().data.shape().to_vec(), 
137            other.0.borrow().data.shape().to_vec()
138        );
139        op.forward(&[self, other])
140    }
