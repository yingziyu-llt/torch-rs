1use super::tensor::Tensor;
2use ndarray::ArrayD;
3use std::{collections::HashSet, rc::Rc};
4
5impl Tensor {
6    /// 反向传播：从当前张量出发，递归地计算所有依赖张量的梯度。
7    ///
8    /// 通常用于loss.backward()。
9    pub fn backward(&self) {
10        // 1. 初始化自身的梯度为全1（通常用于标量loss）
11        {
12            let mut self_data = self.0.borrow_mut();
13            self_data.grad = Some(ArrayD::ones(self_data.data.shape()));
14        }
15
16        // 2. 拓扑排序，确保每个节点在所有子节点之后被处理
17        fn topo_sort(tensor: &Tensor, visited: &mut HashSet<*const ()>, order: &mut Vec<Tensor>) {
18            let ptr = Rc::as_ptr(&tensor.0) as *const ();
19            if visited.contains(&ptr) {
20                return;
21            }
22            visited.insert(ptr);
23            let parents = tensor.0.borrow().parents.clone();
24            for parent_weak in parents {
25                let parent_rc = parent_weak.clone();
26                topo_sort(&Tensor(parent_rc), visited, order);
27            }
28            order.push(tensor.clone());
29        }
30
31        let mut visited = HashSet::new();
32        let mut topo_order = Vec::new();
33        topo_sort(self, &mut visited, &mut topo_order);
34
35        // 3. 反向遍历拓扑序，执行梯度传播
36        for tensor in topo_order.into_iter().rev() {
37            let creator = tensor.0.borrow().creator.clone();
38            //println!("Processing tensor: {:?}", tensor);
39            if let Some(op) = creator {
40                let grads = op.backward(&tensor);
41                let parents = tensor.0.borrow().parents.clone();
42                for (parent_weak, grad) in parents.into_iter().zip(grads.into_iter()) {
43                    let parent_rc = parent_weak.clone();
44                    let mut parent_data = parent_rc.borrow_mut();
45                    match &mut parent_data.grad {
46                        Some(parent_grad) => {
47                            *parent_grad += &grad;
48                        }
49                        None => {
50                            parent_data.grad = Some(grad);
51                        }
52                    }
53                }
54            }
55        }
56    }
57}
